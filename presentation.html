Bridging Traditional AYUSH and Modern Healthcare

An interactive overview of the MedSetu prototype, a digital platform for standardizing and integrating traditional medical records.

The Challenge: A Fragmented System

Traditional AYUSH healthcare in India relies on diverse, often paper-based systems. This lack of a unified digital standard leads to disjointed records, inefficient data sharing, and difficulty integrating with modern global health frameworks like ICD-11.

Backend/server.js

const express = require("express");
const cors = require("cors");
const fs = require("fs");
const path = require("path");
const jwt = require("jsonwebtoken");
const { v4: uuidv4 } = require("uuid");

const app = express();
const PORT = process.env.PORT || 3000;
const SECRET_KEY = process.env.SECRET_KEY || "hackathonSecret123";

app.use(cors());
app.use(express.json());

// Mock data (in-memory database)
const users = require("./data/users.json");
const namasteCodes = require("./data/namaste.json");
let records = require("./data/records.json");
let loginTransactions = require("./data/login-transactions.json");

// Utility to save data (for demonstration purposes)
const saveData = (filename, data) => {
    fs.writeFileSync(path.join(__dirname, "data", filename), JSON.stringify(data, null, 2));
};

// JWT Middleware
const authenticateToken = (req, res, next) => {
    const authHeader = req.headers['authorization'];
    const token = authHeader && authHeader.split(' ')[1];
    if (token == null) return res.sendStatus(401);

    jwt.verify(token, SECRET_KEY, (err, user) => {
        if (err) return res.sendStatus(403);
        req.user = user;
        next();
    });
};

// Routes
app.post("/api/login", (req, res) => {
    const { username, password } = req.body;
    const user = users.find(u => u.username === username && u.password === password);
    if (user) {
        const token = jwt.sign(user, SECRET_KEY, { expiresIn: '1h' });
        loginTransactions.push({ userId: user.id, timestamp: new Date().toISOString() });
        saveData("login-transactions.json", loginTransactions);
        res.json({ token, user });
    } else {
        res.status(401).json({ message: "Invalid credentials" });
    }
});

app.get("/api/records", authenticateToken, (req, res) => {
    const { role, id } = req.user;
    if (role === "doctor") {
        res.json(records.filter(rec => rec.doctor_id === id));
    } else if (role === "patient") {
        res.json(records.filter(rec => rec.patient_id === id));
    } else {
        res.status(403).json({ message: "Forbidden" });
    }
});

app.post("/api/records", authenticateToken, (req, res) => {
    const { patient_id, namaste_code, note } = req.body;
    if (req.user.role !== "doctor") return res.status(403).json({ message: "Forbidden" });

    const newRecord = {
        id: uuidv4(),
        patient_id,
        doctor_id: req.user.id,
        namaste_code,
        note,
        createdAt: new Date().toISOString()
    };
    records.push(newRecord);
    saveData("records.json", records);
    res.status(201).json(newRecord);
});

app.get("/api/namaste", (req, res) => {
    res.json(namasteCodes);
});

app.listen(PORT, () => {
    console.log(Server running on port ${PORT});
});

Frontend/js/app.js

const API_URL = "http://localhost:3000";

document.addEventListener("DOMContentLoaded", () => {
    // Basic search functionality for the home page
    const searchBox = document.getElementById('searchBox');
    if (searchBox) {
        searchBox.addEventListener('input', async (e) => {
            const query = e.target.value;
            const resultsList = document.getElementById('results');
            resultsList.innerHTML = '';
            if (query.length > 2) {
                const response = await fetch(${API_URL}/api/namaste?q=${query});
                const data = await response.json();
                data.forEach(item => {
                    const li = document.createElement('li');
                    li.textContent = ${item.code} - ${item.display};
                    resultsList.appendChild(li);
                });
            }
        });
    }

    // Login Form logic
    const loginForm = document.getElementById('loginForm');
    if (loginForm) {
        loginForm.addEventListener('submit', async (e) => {
            e.preventDefault();
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            try {
                const response = await fetch(${API_URL}/api/login, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ username, password })
                });
                const data = await response.json();
                if (response.ok) {
                    localStorage.setItem('token', data.token);
                    localStorage.setItem('role', data.user.role);
                    window.location.href = data.user.role === 'doctor' ? 'doctor.html' : 'patient.html';
                } else {
                    console.error('Login failed:', data.message);
                }
            } catch (error) {
                console.error('Error:', error);
            }
        });
    }
});

Frontend/js/doctor.js

const API_URL = "http://localhost:3000";
const token = localStorage.getItem("token");

async function loadPatients() {
    if (!token) {
        window.location.href = "login.html";
        return;
    }
    const response = await fetch(${API_URL}/api/records, {
        headers: { 'Authorization': Bearer ${token} }
    });
    const records = await response.json();
    const patientListDiv = document.getElementById('patientList');
    if (patientListDiv) {
        patientListDiv.innerHTML = '
Patient Records
';
        records.forEach(rec => {
            const p = document.createElement('p');
            p.textContent = ID: ${rec.patient_id}, Code: ${rec.namaste_code}, Notes: ${rec.note};
            patientListDiv.appendChild(p);
        });
    }

    const saveRecordBtn = document.getElementById('saveRecord');
    if (saveRecordBtn) {
        saveRecordBtn.addEventListener('click', async () => {
            const patientId = document.getElementById('patientId').value;
            const namasteCode = document.getElementById('namasteCode').value;
            const note = document.getElementById('note').value;

            await fetch(${API_URL}/api/records, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Authorization': Bearer ${token}
                },
                body: JSON.stringify({ patient_id: patientId, namaste_code: namasteCode, note })
            });
            loadPatients(); // Reload the list
        });
    }
}

Project Structure

The MedSetu project is organized into a clean and logical directory structure that separates concerns, making the codebase maintainable and scalable.

MedSetu/
├── Backend/          ← code for server (Express, routes, maybe authentication etc.)
├── Frontend/         ← static pages, HTML, JS client-side logic
├── Data/             ← JSON data (terminologies, user / patient / diagnosis records etc.)
├── postman/          ← Postman collection(s) to test API endpoints
├── package.json
├── package-lock.json
└── node_modules/

Future Vision

MedSetu is a foundational prototype. The vision is to expand the platform to include telemedicine consultations, appointment booking, advanced analytics dashboards for public health, and direct integration with insurance provider APIs.
